#include <bits/stdc++.h>

using namespace std;

int matrix[60][60], n, k, t;
bool rs[60][60], cs[60][60], sol;

void method1(int row, int col, int m)
 {
 
 //MAIN LOGIC
    if (row == n && col == n + 1 && m == k && !sol) 
	{
        sol = true;
        cout << "Case #" << t << ": " << "POSSIBLE\n";
		
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= n; ++j) {
                cout << matrix[i][j] << " ";
            }
            cout << "\n";
        }
        return;
    } 
	else if (row > n) 
	{
        return;
    }
	else if (col > n) 
	{
        method1(row + 1, 1, m);
    }
	
    for (int i = 1; i <= n && !sol; ++i) 
	{
        if (!rs[row][i] && !cs[col][i]) 
		{
            rs[row][i] = cs[col][i] = true;
            if (row == col) 
			{
                m += i;
            }
            matrix[row][col] = i;

            method1(row, col + 1, m);

            rs[row][i] = cs[col][i] = false;
            if (row == col) 
			{
                m -= i;
            }
            matrix[row][col] = 0;
        }
    }
}


int main() 
{
    int T;
    scanf(" %d", &T);
    for (t = 1; t <= T; ++t) 
	{
        scanf(" %d %d", &n, &k);
        method1(1, 1, 0);
        if (!sol) 
		{
            cout << "Case #" << t << ": " << "IMPOSSIBLE\n";
        }
        sol = false;
    }
    return 0;
}
